```swift
import Foundation

/// The `data` payload inside a TurnEvent. The backend wraps the agent's onEvent
/// callbacks verbatim, so `type` here matches the original agent event type.
/// Decoded as a dictionary — SwiftUI render code dispatches on `type`.
struct LegacyEventPayload: Codable, Hashable {
    let type: String
    let chunk: String?          // response_chunk
    let content: String?        // thinking
    let tool: String?           // tool_start / tool_result
    let args: AnyCodable?       // tool_start
    let result: String?         // tool_result
    let success: Bool?          // tool_result
    let mediaType: String?      // media
    let path: String?           // media
    let caption: String?        // media
    let task: String?           // subagent_result
}

/// Tiny AnyCodable — just enough to decode tool_start args without exploding on unknown shapes.
struct AnyCodable: Codable, Hashable {
    let value: String  // pretty-printed JSON string

    init(from decoder: Decoder) throws {
        let container = try decoder.singleValueContainer()
        if let s = try? container.decode(String.self) {
            self.value = s
        } else if let d = try? container.decode([String: String].self) {
            self.value = d.map { "\($0.key)=\($0.value)" }.joined(separator: " ")
        } else {
            self.value = ""
        }
    }
    func encode(to encoder: Encoder) throws {
        var c = encoder.singleValueContainer()
        try c.encode(value)
    }
}
```

- [ ] **Step 1: KeychainStore.swift**

```swift
import Foundation
import Security

enum KeychainStore {
    private static let service = "com.regina6.home23.bearer"
    private static let account = "default"

    static func save(token: String) {
        let data = Data(token.utf8)
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrService as String: service,
            kSecAttrAccount as String: account,
        ]
        SecItemDelete(query as CFDictionary)
        var attrs = query
        attrs[kSecValueData as String] = data
        attrs[kSecAttrAccessible as String] = kSecAttrAccessibleAfterFirstUnlock
        SecItemAdd(attrs as CFDictionary, nil)
    }

    static func load() -> String? {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrService as String: service,
            kSecAttrAccount as String: account,
            kSecReturnData as String: true,
            kSecMatchLimit as String: kSecMatchLimitOne,
        ]
        var out: AnyObject?
        guard SecItemCopyMatching(query as CFDictionary, &out) == errSecSuccess,
              let data = out as? Data,
              let s = String(data: data, encoding: .utf8) else { return nil }
        return s
    }

    static func clear() {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrService as String: service,
            kSecAttrAccount as String: account,
        ]
        SecItemDelete(query as CFDictionary)
    }
}
```

```swift
import Foundation

struct HostConfig: Codable, Equatable {
    let baseURL: URL
    // bearer token retrieved separately via KeychainStore

    private static let defaultsKey = "home23.hostConfig"

    static func loadFromDefaults() -> HostConfig? {
        guard let data = UserDefaults.standard.data(forKey: defaultsKey),
              let cfg = try? JSONDecoder().decode(HostConfig.self, from: data) else { return nil }
        return cfg
    }

    func persist() {
        guard let data = try? JSONEncoder().encode(self) else { return }
        UserDefaults.standard.set(data, forKey: HostConfig.defaultsKey)
    }

    static func clear() {
        UserDefaults.standard.removeObject(forKey: defaultsKey)
        KeychainStore.clear()
    }

    /// Current bearer token from keychain, if any.
    var bearerToken: String? { KeychainStore.load() }
}
```

```swift
enum APIError: LocalizedError, Equatable {
```

```swift
import Foundation

struct Endpoint {
    enum Method: String { case GET, POST, DELETE }

    let method: Method
    let path: String
    let query: [URLQueryItem]

    init(_ method: Method, _ path: String, query: [URLQueryItem] = []) {
        self.method = method
        self.path = path
        self.query = query
    }

    func url(base: URL) throws -> URL {
        guard var comps = URLComponents(url: base.appendingPathComponent(path), resolvingAgainstBaseURL: false) else {
            throw APIError.invalidURL
        }
        if !query.isEmpty {
            comps.queryItems = (comps.queryItems ?? []) + query
        }
        guard let u = comps.url else { throw APIError.invalidURL }
        return u
    }
}
```

```swift
import Foundation
import os.log

/// Bearer-auth URLSession wrapper against a per-agent bridge port or the dashboard port.
final class APIClient {
    private let session: URLSession
    private let logger = Logger(subsystem: "com.regina6.home23", category: "APIClient")

    init(session: URLSession = .shared) {
        self.session = session
    }

    func send<Response: Decodable>(
        _ endpoint: Endpoint,
        base: URL,
        bearerToken: String?,
        responseType: Response.Type = Response.self
    ) async throws -> Response {
        let req = try makeRequest(endpoint, base: base, bearerToken: bearerToken, body: Optional<EmptyBody>.none)
        return try await perform(req)
    }

    func sendBody<Body: Encodable, Response: Decodable>(
        _ endpoint: Endpoint,
        base: URL,
        bearerToken: String?,
        body: Body,
        responseType: Response.Type = Response.self
    ) async throws -> Response {
        let req = try makeRequest(endpoint, base: base, bearerToken: bearerToken, body: body)
        return try await perform(req)
    }

    @discardableResult
    func sendVoid(_ endpoint: Endpoint, base: URL, bearerToken: String?) async throws -> Data {
        let req = try makeRequest(endpoint, base: base, bearerToken: bearerToken, body: Optional<EmptyBody>.none)
        let (data, resp) = try await session.data(for: req)
        try throwIfError(data: data, response: resp)
        return data
    }

    // MARK: - Helpers

    private struct EmptyBody: Encodable {}

    private func makeRequest<Body: Encodable>(
        _ endpoint: Endpoint,
        base: URL,
        bearerToken: String?,
        body: Body?
    ) throws -> URLRequest {
        let url = try endpoint.url(base: base)
        var req = URLRequest(url: url)
        req.httpMethod = endpoint.method.rawValue
        req.timeoutInterval = 30
        if let t = bearerToken, !t.isEmpty {
            req.setValue("Bearer \(t)", forHTTPHeaderField: "Authorization")
        }
        if let body = body, !(body is EmptyBody) {
            req.setValue("application/json", forHTTPHeaderField: "Content-Type")
            req.httpBody = try JSONEncoder().encode(body)
        }
        return req
    }

    private func perform<Response: Decodable>(_ req: URLRequest) async throws -> Response {
        let (data, resp): (Data, URLResponse)
        do { (data, resp) = try await session.data(for: req) }
        catch { throw APIError.network(underlying: error.localizedDescription) }

        try throwIfError(data: data, response: resp)

        do { return try JSONDecoder().decode(Response.self, from: data) }
        catch { throw APIError.decoding(message: error.localizedDescription) }
    }

    private func throwIfError(data: Data, response: URLResponse) throws {
        guard let http = response as? HTTPURLResponse else { return }
        switch http.statusCode {
        case 200..<300: return
        case 401: throw APIError.unauthorized
        case 404: throw APIError.notFound
        case 409: throw APIError.conflict(body: data)
        default: throw APIError.server(status: http.statusCode, body: data)
        }
    }
}
```

```swift
import Foundation

/// Record delivered by the stream. Either an event or a final turn envelope.
enum TurnStreamRecord {
    case event(TurnEvent)
    case envelope(TurnEnvelope)
    case done
}

/// AsyncSequence that yields TurnStreamRecord values as they arrive over SSE.
/// Uses URLSession.bytes for line-by-line streaming with cooperative cancellation.
struct TurnStream: AsyncSequence {
    typealias Element = TurnStreamRecord

    let url: URL
    let bearerToken: String?

    struct AsyncIterator: AsyncIteratorProtocol {
        var lineIterator: URLSession.AsyncBytes.AsyncIterator
        private let decoder: JSONDecoder

        init(lineIterator: URLSession.AsyncBytes.AsyncIterator, decoder: JSONDecoder) {
            self.lineIterator = lineIterator
            self.decoder = decoder
        }

        mutating func next() async throws -> TurnStreamRecord? {
            var buffer = ""
            // SSE: events are separated by a blank line; each event is one or more "field: value" lines.
            // We only care about `data:` lines here.
            while true {
                guard let byte = try await lineIterator.next() else { return nil }
                if byte == 0x0A { // \n
                    if buffer.isEmpty {
                        // blank line → end of event. If we've accumulated a data payload, parse it.
                        continue
                    }
                    if let payload = Self.extractDataPayload(&buffer) {
                        if payload == "[DONE]" { return .done }
                        if let data = payload.data(using: .utf8) {
                            if let ev = try? decoder.decode(TurnEvent.self, from: data) {
                                return .event(ev)
                            }
                            if let env = try? decoder.decode(TurnEnvelope.self, from: data) {
                                return .envelope(env)
                            }
                        }
                    }
                    buffer = ""
                } else {
                    buffer.append(Character(UnicodeScalar(byte)))
                }
            }
        }

        /// Extract a `data: X` value. SSE spec supports multi-line data concatenated with \n;
        /// home23 emits single-line data so we just peel the prefix.
        static func extractDataPayload(_ line: inout String) -> String? {
            let prefix = "data: "
            if line.hasPrefix(prefix) {
                return String(line.dropFirst(prefix.count))
            }
            if line.hasPrefix(":") { return nil } // comment/heartbeat
            return nil
        }
    }

    func makeAsyncIterator() -> AsyncIterator {
        var req = URLRequest(url: url)
        req.setValue("text/event-stream", forHTTPHeaderField: "Accept")
        if let t = bearerToken, !t.isEmpty {
            req.setValue("Bearer \(t)", forHTTPHeaderField: "Authorization")
        }
        req.timeoutInterval = 60 * 60 // 1h — streams are long-lived
        // Kick off bytes request; AsyncBytes will stream as lines arrive.
        // Synchronous creation of the bytes is fine — URLSession.bytes(for:) is non-blocking under the hood.
        // Note: this init can throw; swallow into an empty sequence on failure (consumer will see nil on next()).
        let bytes = URLSession.shared.bytes(for: req)
        // bytes is an async (URLSession.AsyncBytes, URLResponse) tuple — can't resolve sync.
        // Workaround: use a custom runner that awaits the tuple before returning iteration control.
        // Pragmatic: wrap the await in a Task and block via a semaphore-free mechanism using a Continuation.
        //
        // For simplicity, we return an iterator that lazily awaits on first .next() — implemented below.
        return DelayedIterator(request: req, decoder: JSONDecoder()).makeIterator()
    }

    /// Wraps an async URLSession.bytes(for:) call, awaiting it on first `.next()`.
    private struct DelayedIterator {
        let request: URLRequest
        let decoder: JSONDecoder

        func makeIterator() -> AsyncIterator {
            // We need an iterator synchronously. Cheat: create an iterator over a stream we'll fill.
            // The cleanest implementation uses AsyncThrowingStream.
            let stream = AsyncThrowingStream<UInt8, Error> { continuation in
                Task {
                    do {
                        let (bytes, _) = try await URLSession.shared.bytes(for: request)
                        for try await b in bytes {
                            continuation.yield(b)
                        }
                        continuation.finish()
                    } catch {
                        continuation.finish(throwing: error)
                    }
                }
            }
            // Adapter: convert AsyncThrowingStream<UInt8> → URLSession.AsyncBytes.AsyncIterator lookalike.
            // We need a line reader. Simpler: reimplement the .next() logic directly here.
            return AsyncIterator(stream: stream, decoder: decoder)
        }
    }
}
```

```swift
enum TurnStreamRecord {
    case event(TurnEvent)
    case envelope(TurnEnvelope)
    case done
}

struct TurnStream: AsyncSequence {
    typealias Element = TurnStreamRecord

    let url: URL
    let bearerToken: String?

    func makeAsyncIterator() -> Iterator {
        Iterator(url: url, bearerToken: bearerToken)
    }

    final class Iterator: AsyncIteratorProtocol {
        private let url: URL
        private let bearerToken: String?
        private var task: URLSessionDataTask?
        private var byteStream: AsyncThrowingStream<UInt8, Error>.Iterator?
        private let decoder = JSONDecoder()

        init(url: URL, bearerToken: String?) {
            self.url = url
            self.bearerToken = bearerToken
        }

        private func startIfNeeded() async throws {
            guard byteStream == nil else { return }
            var req = URLRequest(url: url)
            req.setValue("text/event-stream", forHTTPHeaderField: "Accept")
            if let t = bearerToken, !t.isEmpty {
                req.setValue("Bearer \(t)", forHTTPHeaderField: "Authorization")
            }
            req.timeoutInterval = 60 * 60
            let (bytes, _) = try await URLSession.shared.bytes(for: req)

            let stream = AsyncThrowingStream<UInt8, Error> { continuation in
                Task {
                    do {
                        for try await b in bytes { continuation.yield(b) }
                        continuation.finish()
                    } catch {
                        continuation.finish(throwing: error)
                    }
                }
            }
            self.byteStream = stream.makeAsyncIterator()
        }

        func next() async throws -> TurnStreamRecord? {
            try await startIfNeeded()
            guard var it = byteStream else { return nil }

            var line = ""
            while true {
                guard let b = try await it.next() else {
                    byteStream = nil
                    return nil
                }
                if b == 0x0A {
                    defer { line = "" }
                    if line.isEmpty { continue }
                    if line.hasPrefix(":") { continue } // heartbeat
                    if !line.hasPrefix("data: ") { continue }
                    let payload = String(line.dropFirst("data: ".count))
                    if payload == "[DONE]" {
                        byteStream = it
                        return .done
                    }
                    guard let d = payload.data(using: .utf8) else { continue }
                    if let ev = try? decoder.decode(TurnEvent.self, from: d) {
                        byteStream = it
                        return .event(ev)
                    }
                    if let env = try? decoder.decode(TurnEnvelope.self, from: d) {
                        byteStream = it
                        return .envelope(env)
                    }
                    continue
                }
                line.append(Character(UnicodeScalar(b)))
            }
        }
    }
}
```

```swift
@MainActor
final class ConversationCache {
    let db: DatabaseQueue

    init() throws {
        let appSupport = try FileManager.default.url(for: .applicationSupportDirectory, in: .userDomainMask, appropriateFor: nil, create: true)
        let path = appSupport.appendingPathComponent("home23-cache.sqlite").path
        self.db = try DatabaseQueue(path: path)
        try CacheSchema.migrate(db)
    }

    // MARK: - Conversations

    func upsertConversation(id: String, agent: String, title: String?) throws {
        let now = Date().timeIntervalSince1970
        try db.write { db in
            try db.execute(sql: """
                INSERT INTO conversations(id, agent, title, updated_at) VALUES(?, ?, ?, ?)
                ON CONFLICT(id) DO UPDATE SET title = COALESCE(excluded.title, title), updated_at = excluded.updated_at
            """, arguments: [id, agent, title, Int(now)])
        }
    }

    func listConversations(agent: String) throws -> [Conversation] {
        try db.read { db in
            let rows = try Row.fetchAll(db, sql: "SELECT id, agent, title, updated_at FROM conversations WHERE agent = ? ORDER BY updated_at DESC LIMIT 100", arguments: [agent])
            return rows.map {
                Conversation(
                    id: $0["id"],
                    agent: $0["agent"],
                    title: $0["title"],
                    updatedAt: Date(timeIntervalSince1970: TimeInterval($0["updated_at"] as Int64))
                )
            }
        }
    }

    // MARK: - Messages

    func appendMessage(convId: String, role: String, content: String, turnId: String?) throws {
        try db.write { db in
            try db.execute(sql: "INSERT INTO messages(conv_id, role, content, ts, turn_id) VALUES(?, ?, ?, ?, ?)",
                           arguments: [convId, role, content, Int(Date().timeIntervalSince1970), turnId])
        }
    }

    func loadMessages(convId: String, limit: Int = 500) throws -> [(role: String, content: String, ts: Int64, turnId: String?)] {
        try db.read { db in
            let rows = try Row.fetchAll(db, sql: "SELECT role, content, ts, turn_id FROM messages WHERE conv_id = ? ORDER BY ts ASC, id ASC LIMIT ?", arguments: [convId, limit])
            return rows.map { ($0["role"], $0["content"], $0["ts"], $0["turn_id"]) }
        }
    }

    // MARK: - Turns + events

    func upsertTurn(turnId: String, convId: String, status: String, lastSeq: Int, startedAt: Date, endedAt: Date?) throws {
        try db.write { db in
            try db.execute(sql: """

                ON CONFLICT(turn_id) DO UPDATE SET status = excluded.status, last_seq = MAX(last_seq, excluded.last_seq), ended_at = COALESCE(excluded.ended_at, ended_at)

    }

    func reset() throws {
```

```swift
        }

    private var bridge: URL { host.bridgeURL(for: agent) }

    // MARK: - Public actions

    func loadHistory() async {
        // Try cache first
        if let c = cache, let cached = try? c.loadMessages(convId: chatId) {
            messages = cached.map { row in
                switch row.role {
                case "user": return UIMessage(kind: .user, text: row.content)
                default:     return UIMessage(kind: .assistant, text: row.content)
                }
            }
        }
        // Then server history (window)
        struct HistoryResponse: Decodable {
            let records: [HistoryRecord]
        }
        struct HistoryRecord: Decodable {
            let type: String?
            let role: String?
            let content: AnyContent?
            // Ignore tool records, session boundaries, turn records here — cache will be the full store later.
        }
        enum AnyContent: Decodable {
            case string(String); case unknown
            init(from d: Decoder) throws {
                let c = try d.singleValueContainer()
                if let s = try? c.decode(String.self) { self = .string(s) } else { self = .unknown }
            }
            var text: String? { if case .string(let s) = self { return s }; return nil }
        }

        do {
            let resp: HistoryResponse = try await client.send(
                Endpoint(.GET, "/api/chat/history", query: [URLQueryItem(name: "chatId", value: chatId), URLQueryItem(name: "limit", value: "100")]),
                base: bridge,
                bearerToken: host.bearerToken
            )
            let server = resp.records.compactMap { r -> UIMessage? in
                guard r.type == nil, let role = r.role, let t = r.content?.text else { return nil }
                return UIMessage(kind: role == "user" ? .user : .assistant, text: t)
            }
            if !server.isEmpty { messages = server }
        } catch {
            // Show cache only — banner if nothing to show
            if messages.isEmpty { banner = "Can't reach home23 — no cached history." }
        }

        await resumePendingIfAny()
    }

    func send() async {
        let text = composer.trimmingCharacters(in: .whitespaces)
        guard !text.isEmpty, !isStreaming else { return }
        composer = ""

        messages.append(UIMessage(kind: .user, text: text))
        try? cache?.appendMessage(convId: chatId, role: "user", content: text, turnId: nil)

        isStreaming = true
        activeCursor = -1

        struct StartBody: Encodable { let chatId: String; let message: String }
        do {
            let resp: StartTurnResponse = try await client.sendBody(
                Endpoint(.POST, "/api/chat/turn"),
                base: bridge,
                bearerToken: host.bearerToken,
                body: StartBody(chatId: chatId, message: text)
            )
            activeTurnId = resp.turn_id
            openStream(cursor: -1)
        } catch APIError.conflict(let bodyData) {
            if let bodyData, let existing = try? JSONDecoder().decode([String: String].self, from: bodyData), let id = existing["turn_id"] {
                activeTurnId = id
                openStream(cursor: -1)
            } else {
                finalize(withError: "Turn start conflict")
            }
        } catch {
            finalize(withError: (error as? APIError)?.errorDescription ?? error.localizedDescription)
        }
    }

    func stop() {
        guard let turnId = activeTurnId else { return }
        Task {
            struct StopBody: Encodable { let chatId: String; let turn_id: String }
            _ = try? await client.sendBody(
                Endpoint(.POST, "/api/chat/stop-turn"),
                base: bridge,
                bearerToken: host.bearerToken,
                body: StopBody(chatId: chatId, turn_id: turnId),
                responseType: [String: Bool].self
            )
        }
    }

    func reconnectIfNeeded() {
        guard isStreaming, activeTurnId != nil, streamTask == nil else { return }
        openStream(cursor: activeCursor)
    }

    // MARK: - Internal

    private func resumePendingIfAny() async {
        do {
            let resp: PendingTurnsResponse = try await client.send(
                Endpoint(.GET, "/api/chat/pending", query: [URLQueryItem(name: "chatId", value: chatId)]),
                base: bridge,
                bearerToken: host.bearerToken
            )
            if let turn = resp.pending.last {
                activeTurnId = turn.turn_id
                activeCursor = -1
                isStreaming = true
                openStream(cursor: -1)
            }
        } catch { /* silent */ }
    }

    private func openStream(cursor: Int) {
        streamTask?.cancel()
        guard let turnId = activeTurnId else { return }
        let url = bridge.appendingPathComponent("api/chat/stream")
        var comps = URLComponents(url: url, resolvingAgainstBaseURL: false)!
        comps.queryItems = [
            URLQueryItem(name: "chatId", value: chatId),
            URLQueryItem(name: "turn_id", value: turnId),
            URLQueryItem(name: "cursor", value: "\(cursor)"),
        ]
        guard let streamURL = comps.url else { finalize(withError: "bad stream URL"); return }

        streamTask = Task { [weak self] in
            guard let self else { return }
            let stream = TurnStream(url: streamURL, bearerToken: host.bearerToken)
            do {
                for try await record in stream {
                    if Task.isCancelled { break }
                    switch record {
                    case .event(let ev):
                        self.activeCursor = ev.seq
                        self.apply(ev)
                        try? self.cache?.appendEvent(turnId: ev.turn_id, seq: ev.seq, kind: ev.kind, dataJson: (try? String(data: JSONEncoder().encode(ev.data), encoding: .utf8)) ?? "")
                    case .envelope(let env):
                        if env.status != "pending" {
                            self.finalize(envelope: env)
                            return
                        }
                    case .done:
                        self.finalize(envelope: nil)
                        return
                    }
                }
            } catch {
                // Stream died unexpectedly — leave activeTurnId set so reconnectIfNeeded() can retry.
                self.streamTask = nil
            }
        }
    }

    private func apply(_ ev: TurnEvent) {
        let data = ev.data
        switch data.type {
        case "response_chunk", "text":
            let chunk = data.chunk ?? ""
            if case .assistant = messages.last?.kind {
                messages[messages.count - 1].text += chunk
            } else {
                messages.append(UIMessage(kind: .assistant, text: chunk))
            }
        case "thinking":
            messages.append(UIMessage(kind: .thinking, text: data.content ?? ""))
        case "tool_start":
            messages.append(UIMessage(kind: .tool, text: "", toolName: data.tool, toolArgs: data.args?.value))
        case "tool_result":
            if let i = messages.lastIndex(where: { $0.kind == .tool && $0.toolName == data.tool && $0.toolResult == nil }) {
                messages[i].toolResult = data.result
                messages[i].toolSuccess = data.success ?? true
            }
        default:
            break
        }
    }

    private func finalize(envelope: TurnEnvelope?) {
        isStreaming = false
        if let last = messages.last, case .assistant = last.kind {
            try? cache?.appendMessage(convId: chatId, role: "assistant", content: last.text, turnId: activeTurnId)
        }
        activeTurnId = nil
        activeCursor = -1
        streamTask = nil
        if let env = envelope {
            if env.status == "error" { banner = env.error ?? "Error" }
            else if env.status == "orphaned" { banner = "Previous turn interrupted — try resending." }
        }
    }

    private func finalize(withError text: String) {
        isStreaming = false
        banner = text
        streamTask = nil
        activeTurnId = nil
    }
}
```

## Known plan fragilities the engineer should watch for

- **Agent list endpoint shape** (`/home23/api/settings/agents`) — Task 7 Step 1 verifies with curl before coding. If the shape differs, update `AgentListResponse` in Task 4.
- **Conversation list endpoint** (`/api/chat/conversations`) — the backend plan Task 8 adds this; verify with curl before Task 12.
- **Env (sandbox vs production) mismatch** — the most common APNs failure. Always cross-check `device-registry.json` env vs build type.
- **iOS 17 vs 16** — TurnStream uses `AsyncThrowingStream` + `URLSession.bytes(for:)`; both require iOS 15+. Targeting iOS 17 is fine.
- **Keychain persistence across builds** — dev-signed + TestFlight have different Keychain access groups sometimes. Re-onboarding after switching between them is expected.
- **Conversation list title** — server returns `null` for title; Task 12 uses "Conversation" as fallback. Client should compute title from first user message (future polish).
